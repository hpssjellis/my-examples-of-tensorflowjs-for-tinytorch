/*
 * ESP32-S3 Data Collection - Complete Version
 * Captures 96x96 images for SD storage and OLED preview.
 */

#include "esp_camera.h"
#include "FS.h"
#include "SD.h"
#include "SPI.h"
#include <U8g2lib.h>
#include <Wire.h>

// --- Configuration ---
const int myCaptureSize = 96;
const int myOledThreshold = 120; // Brightness for binary OLED view

// Pin Definitions
#define PWDN_GPIO_NUM     -1
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM     10
#define SIOD_GPIO_NUM     40
#define SIOC_GPIO_NUM     39
#define Y9_GPIO_NUM       48
#define Y8_GPIO_NUM       11
#define Y7_GPIO_NUM       12
#define Y6_GPIO_NUM       14
#define Y5_GPIO_NUM       16
#define Y4_GPIO_NUM       18
#define Y3_GPIO_NUM       17
#define Y2_GPIO_NUM       15
#define VSYNC_GPIO_NUM    38
#define HREF_GPIO_NUM     47
#define PCLK_GPIO_NUM     13
#define LED_BUILTIN       21

U8G2_SSD1306_72X40_ER_1_HW_I2C u8g2(U8G2_R2, u8x8_pin_none); 

bool camera_sign = false;          
bool sd_sign = false;              
int imageCount = 1;

void myUpdateOled(const char* msg, int count = -1) {
    u8g2.firstPage();
    do {
        u8g2.setFont(u8g2_font_ncenB08_tr);
        u8g2.setCursor(0, 15);
        u8g2.print(msg);
        if (count != -1) {
            u8g2.setCursor(0, 30);
            u8g2.print("Count: "); u8g2.print(count);
        }
    } while (u8g2.nextPage());
}

void mySavePhoto() {
    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb) return;

    char myPath[32];
    sprintf(myPath, "/images/img_%d.jpg", imageCount);
    
    File myFile = SD.open(myPath, FILE_WRITE);
    if (myFile) {
        myFile.write(fb->buf, fb->len);
        myFile.close();
        Serial.printf("Saved: %s (%d bytes)\n", myPath, fb->len);
        imageCount++;
    }

    // OLED Preview logic
    uint8_t *myRgb = (uint8_t *)ps_malloc(fb->width * fb->height * 3);
    if (myRgb && fmt2rgb888(fb->buf, fb->len, fb->format, myRgb)) {
        int oW = u8g2.getDisplayWidth();
        int oH = u8g2.getDisplayHeight();
        u8g2.firstPage();
        do {
            for (int y = 0; y < oH; y++) {
                for (int x = 0; x < oW; x++) {
                    int ix = (x * fb->width) / oW;
                    int iy = (y * fb->height) / oH;
                    int idx = (iy * fb->width + ix) * 3;
                    uint8_t lum = (myRgb[idx] + myRgb[idx+1] + myRgb[idx+2]) / 3;
                    if (lum > myOledThreshold) u8g2.drawPixel(x, y);
                }
            }
            u8g2.setFont(u8g2_font_04b_03_tr);
            u8g2.setCursor(0, 8); u8g2.print("REC");
        } while (u8g2.nextPage());
    }
    free(myRgb);
    esp_camera_fb_return(fb);
}

void setup() {
    Serial.begin(115200);
    pinMode(A0, INPUT);
    pinMode(LED_BUILTIN, OUTPUT);
    u8g2.begin();

    camera_config_t config;
    config.ledc_channel = LEDC_CHANNEL_0;
    config.ledc_timer = LEDC_TIMER_0;
    config.pin_d0 = Y2_GPIO_NUM;
    config.pin_d1 = Y3_GPIO_NUM;
    config.pin_d2 = Y4_GPIO_NUM;
    config.pin_d3 = Y5_GPIO_NUM;
    config.pin_d4 = Y6_GPIO_NUM;
    config.pin_d5 = Y7_GPIO_NUM;
    config.pin_d6 = Y8_GPIO_NUM;
    config.pin_d7 = Y9_GPIO_NUM;
    config.pin_xclk = XCLK_GPIO_NUM;
    config.pin_pclk = PCLK_GPIO_NUM;
    config.pin_vsync = VSYNC_GPIO_NUM;
    config.pin_href = HREF_GPIO_NUM;
    config.pin_sscb_sda = SIOD_GPIO_NUM;
    config.pin_sscb_scl = SIOC_GPIO_NUM;
    config.pin_pwdn = PWDN_GPIO_NUM;
    config.pin_reset = RESET_GPIO_NUM;
    config.xclk_freq_hz = 20000000;
    config.frame_size = FRAMESIZE_96X96;
    config.pixel_format = PIXFORMAT_JPEG;
    config.fb_location = CAMERA_FB_IN_PSRAM;
    config.jpeg_quality = 10;
    config.fb_count = 2;

    if (esp_camera_init(&config) == ESP_OK) camera_sign = true;
    if (SD.begin(21)) {
        sd_sign = true;
        SD.mkdir("/images");
    }

    if (camera_sign && sd_sign) myUpdateOled("System Ready");
    else myUpdateOled("Error Init");
}

void loop() {
    if (analogRead(A0) > 2000) { // Trigger on button/pin high
        digitalWrite(LED_BUILTIN, LOW);
        mySavePhoto();
        delay(300);
        digitalWrite(LED_BUILTIN, HIGH);
    }
}
